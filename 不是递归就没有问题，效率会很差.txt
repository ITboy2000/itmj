T0 = 0, T1 = 1, T2 = 1, 且在 n >= 0?的条件下 Tn+3 = Tn + Tn+1 + Tn+2

class Solution {
    public int tribonacci(int n) {
        if(n==0){
            return 0;
        }
        if(n==1||n==2){
            return 1;
        }
        int i1=0;
        int i2=1;
        int i3=1;
        for(int i=4;i<=n;i++){
            int temp=i1+i2+i3;
            i1=i2;
            i2=i3;
            i3=temp;
        }
        return i1+i2+i3;
    }
}
